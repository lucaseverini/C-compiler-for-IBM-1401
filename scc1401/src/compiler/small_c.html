/** Main production. */
program ::= ( globalVariableDeclaration | functionDeclaration | functionDefinition )*
bareType ::= ( "char" | "int" )
Type ::= ( bareType ( "*" )* )
globalVariableDeclaration ::= ( bareType )? variableDeclarationList ";"
localVariableDeclaration ::= bareType variableDeclarationList ";"
variableDeclarationList ::= variableInitializer ( "," variableInitializer )*
variableInitializer ::= suffixedVariable ( "=" ternaryExpression )? 
//here is where we add it to our SymbolTable, since we know
//the full type
suffixedVariable ::= ( PrefixedVariable "[" ( expression ) "]" | PrefixedVariable )
PrefixedVariable ::= ( Identifier | "*" PrefixedVariable )
functionDeclaration ::= Type Identifier ( "(" ( FunctionDeclarationParameter ( "," FunctionDeclarationParameter )* )? ")" ) ";"
FunctionDeclarationParameter ::= ( Type ( Identifier )? | Identifier ) ( "[" ( expression )? "]" )?
functionDefinition ::= ( Type )? Identifier ( ( "(" ( functionDefinitionParameter ( "," functionDefinitionParameter )* )? ( "," "..." )? ")" ) ) blockStatement
functionDefinitionParameter ::= ( Type )? Identifier ( "[" ( expression )? "]" )?
blockStatement ::= "{" ( localVariableDeclaration )* ( statement )* "}"
statement ::= ";"
| expression ";"
| blockStatement
| "if" "(" expression ")" statement ( "else" statement )?
| "while" "(" expression ")" statement
| "for" "(" ( expression ";" | ";" ) ( expression ";" | ";" ) ( expression ")" | ")" ) statement
| "do" statement "while" "(" expression ")" ";"
| "return" ( expression )? ";"

/** Left-to-right*/
expression ::= assignmentExpression ( "," assignmentExpression )* 
/** Right-to-left */
assignmentExpression ::= ( ternaryExpression ( "+=" assignmentExpression | "-=" assignmentExpression | "*=" assignmentExpression | "/=" assignmentExpression | "%=" assignmentExpression | ">>=" assignmentExpression | "<<=" assignmentExpression | "&=" assignmentExpression | "^=" assignmentExpression | "|=" assignmentExpression | "=" assignmentExpression )? )
ternaryExpression ::= ( orExpression ( "?" expression ":" ternaryExpression )? ) 
/** left-to-right*/
orExpression ::= andExpression ( "||" andExpression )*
andExpression ::= bitOrExpression ( "&&" bitOrExpression )*
bitOrExpression ::= bitXorExpression ( "|" bitXorExpression )*
bitXorExpression ::= bitAndExpression ( "^" bitAndExpression )*
bitAndExpression ::= equalityExpression ( "&" equalityExpression )*
equalityExpression ::= greaterThanExpression ( "==" greaterThanExpression | "!=" greaterThanExpression )*
greaterThanExpression ::= lessThanExpression ( ">" lessThanExpression | ">=" lessThanExpression )*
lessThanExpression ::= bitshiftExpression ( "<" bitshiftExpression | "<=" bitshiftExpression )*
bitshiftExpression ::= additiveExpression ( "<<" additiveExpression | ">>" additiveExpression )*
additiveExpression ::= multiplicativeExpression ( "+" multiplicativeExpression | "-" multiplicativeExpression )*
multiplicativeExpression ::= ( prefixExpression ( "*" prefixExpression | "/" prefixExpression | "%" prefixExpression )* )
prefixExpression ::= ( suffixExpression | "++" prefixExpression | "--" prefixExpression | "+" prefixExpression | "-" prefixExpression | "!" prefixExpression | "~" prefixExpression | "*" prefixExpression | "&" prefixExpression | "sizeof" prefixExpression | "sizeof" "(" Type ")" | "(" Type ")" prefixExpression )
suffixExpression ::= ( AtomicExpression ( "++" | "--" | "(" expressionList ")" | "[" expression "]" )* )
expressionList ::= ( assignmentExpression ( "," assignmentExpression )* )?
AtomicExpression ::= ( Identifier | Number | StringLiteral | CharacterConstant | "(" expression ")" )
Identifier ::= <IDENTIFIER>
